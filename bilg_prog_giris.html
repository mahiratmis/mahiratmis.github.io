Ekrana Merhaba Dunya yazdırmak

C Kodu 

/* Yorumlarınızı buraya yazabilirsiniz

   h uzantılı dosyalar başlık(header) dosyalarıdır.
   içlerinde çeşitli tanımlamalar bulunur.
   stdio.h dosyasında aşağıda kullandığımız printf fonksiyonu tanımlıdır.
   #include komutu ile stdio.h dosyasında herkesin kullanımına açık
   olan tanımlamaları bizim programın kullanmasını sağlıyoruz */
#include <stdio.h>

/* Program çalışmaya main fonksiyonunun içinden başlar*/
int main()
{ // fonksiyon bloğu başlama
    /* printf ekrana mesaj yazmak için kullanılır*/
    printf("Merhaba Dunya");
    /* main fonksiyonu çalışmasını sorunsuz bir şekilde bitirdiğinde
       yaygın olarak 0 döndürür*/
    return 0;
} // fonksiyon bloğu bitiş

Problem Tanımı

Ekrana özel karakterleri yazdırmak

C Kodu 

#include <stdio.h>

int main()
{   // \n  karakterinden sonra ekranda bir alt satıra geçilir
    printf("Merhaba\nDunya\n");
    // \a  karakteri bilgisayarda alarm sesi çıkmasını sağlar
    printf("Islem bitti\n\a");
    // \t  ekranda yatay olarak bir tab boşluk bırakır
    printf("Merhaba\tDunya\n");
    // \"  ekranda çift tırnak karakterini yazabilmemizi sağlar
    printf("Merhaba \"Dunya\" \n");
    // \?  ekranda soru işareti karakterini yazabilmemizi sağlar
    printf("Merhaba, Nasilsin\? \n");
    // \'  ekranda tek tırnak karakterini yazabilmemizi sağlar
    printf("Merhaba, \'Dunya\' \n");
    // \\  ekranda ters bölü karakterini yazabilmemizi sağlar
    printf("Merkez\\YALOVA");

    return 0;
}

Ekran Çıktısı

Dunya

Merhaba "Dunya"
Merhaba, Nasilsin?
Merhaba, 'Dunya'
Merkez\YALOVA

Farklı Türlerde değişken tanımlama

C Kodu 

#include <stdio.h>

/* Değişkenler içerisinde değer sakladığımız bellek birimleridir.
   Değişken tanımlarken dikkat edilmesi gerekenler
   1. Türkçe karakter kullanmayın
   2. Değişken adına rakamla başlamayın
   3. Değişken adlarında boşluk kullanmayın
*/
int main(){
    /* Değişken tanımlama */
    // karakter türünde değişken tanımla
    char harf;
    // küçük tamsayı türünde değişken tanımlama
    short yas;
    // tamsayı türünde değişken tanımlama
    int dogum_tarihi;
    // büyük tamsayı türünde değişken tanımlama
    long tc_no;
    // küçük ondalıklı sayı türünde değişken tanımlama
    float boy;
    // büyük ondalıklı sayı türünde değişken tanımlama
    double bakiye;

    /* Değişkene değer verme (atama) */
    harf = 'E';
    yas= 29;
    dogum_tarihi= 1987;
    tc_no=19123454209;
    boy=1.83f;
    bakiye=1000000.45;
}

Farklı Veri Türlerinin bellekte kaç Byte yer kapladığını ve Değer Aralıklarını Ekrana Yazdırma

C Kodu 

#include <stdio.h>
#include <limits.h>
#include <float.h>

/* Veri türlerinin bellekte kaç byte yer kapladıklarını
   Alacakları minmum ve maximum değerleri ekrana yazdırma.
*/
int main()
{
  // turlerin bellekte kac byte
  // yer kapladiklarini hesapla
  long byt_chr  = sizeof(char);
  long byt_shrt = sizeof(short);
  long byt_int  = sizeof(int);
  long byt_lng  = sizeof(long);
  long byt_flt  = sizeof(float);
  long byt_dbl  = sizeof(double);

  printf("  Veri Turu       Kapladigi Alan(Byte)\n");
  printf("  ---------       --------------------\n");
  printf("%9s %18ld \n","char", byt_chr);
  printf("%9s %18ld \n","short", byt_shrt);
  printf("%9s %18ld \n","int", byt_int);
  printf("%9s %18ld \n","long", byt_lng);
  printf("%9s %18ld \n","float", byt_flt);
  printf("%9s %18ld \n","double", byt_dbl);
  printf("\n");
  printf("  Veri Turu       Min Deger       Max Deger\n");
  printf("  ---------       ---------       ---------\n");
  printf("%11s %15d %15d\n","char", CHAR_MIN, CHAR_MAX);
  printf("%11s %15d %15d\n","short", SHRT_MIN, SHRT_MAX);
  printf("%11s %15d %15d\n","int", INT_MIN, INT_MAX);
  printf("%11s %15ld %15ld\n","long", LONG_MIN, LONG_MAX);
  printf("%11s %15e %15e\n","float", FLT_MIN , FLT_MAX);
  printf("%11s %15e %15e\n","double", DBL_MIN, DBL_MAX);

  return 0;
}
