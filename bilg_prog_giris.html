C Kodu 

#include <stdio.h>

int main()
{   // \n  karakterinden sonra ekranda bir alt satıra geçilir
    printf("Merhaba\nDunya\n");
    // \a  karakteri bilgisayarda alarm sesi çıkmasını sağlar
    printf("Islem bitti\n\a");
    // \t  ekranda yatay olarak bir tab boşluk bırakır
    printf("Merhaba\tDunya\n");
    // \"  ekranda çift tırnak karakterini yazabilmemizi sağlar
    printf("Merhaba \"Dunya\" \n");
    // \?  ekranda soru işareti karakterini yazabilmemizi sağlar
    printf("Merhaba, Nasilsin\? \n");
    // \'  ekranda tek tırnak karakterini yazabilmemizi sağlar
    printf("Merhaba, \'Dunya\' \n");
    // \\  ekranda ters bölü karakterini yazabilmemizi sağlar
    printf("Merkez\\YALOVA");

    return 0;
}

  printf("%9s %18ld \n","long", byt_lng);

Problem Tanımı

Öğrenciden birinci ve ikinci vize notları istenir. Sonra da geçmek istediği harf notu sorulur.
Program öğrenciye, istediği harf notu ile geçebilmesi için finalden minimum kaç puan alması gerektiğini söyler.
Eğer finalden alması gereken puan 100 ün üzerindeyse bu harf notunu almasının imkansız olduğu söylenir.
Harf notları sırasıyla A,B,C ve D dir. Bu okulda dönem sonu sınav notu ortalaması
vizelerin aritmetik ortalamasın %30 u ile final notunun %70 si toplanarak bulunur.
Öğrencinin;
A ile geçebilmesi için gerekli minimum ortalama 90,
B ile geçebilmesi için gerekli minimum ortalama 70,
C ile geçebilmesi için gerekli minimum ortalama 50,
D ile geçebilmesi için gerekli minimum ortalama 0 dır.

C Kodu 

#include <stdio.h>

int main()
{
 int vize1, vize2, min_ort;
 char harf_notu;
 float final_notu;

 printf("Vize1 ve vize2 notlarini giriniz ");
 scanf("%d%d",&vize1,&vize2);
 printf("Almak istediginiz harf notu [A,B,C,D]");
 scanf(" %c",&harf_notu);

 if(harf_notu =='A')
    min_ort = 90;
 else if (harf_notu =='B')
    min_ort = 70;
 else if (harf_notu =='C')
    min_ort = 50;
 else
    min_ort = 0;

 final_notu = (min_ort - ((vize1+vize2)*0.5f)*0.3f)/0.7f;

 if(final_notu >100)
    printf("Bu harf notunu almaniz imkansiz");
 else
    printf("%f almalisiniz", final_notu);
}

Ekran Çıktısı

 				
Vize1 ve vize2 notlarini giriniz 60 70
Almak istediginiz harf notu [A,B,C,D]A
Bu harf notunu almaniz imkansiz

C Kodu 

#include <stdio.h>

int main()
{
   int sayi,i;
   long long int carpim=1;
   printf("Bir sayi giriniz ");
   scanf("%d",&sayi);
   for(i=2; i<=sayi; ++i)
   {
    carpim *= i;
   }
   printf("%d nin faktoriyeli %lli dir",sayi,carpim);
}

Problem Tanımı

Girilen iki tamsayı değeri arasındaki çift sayıların toplamını ekrana yazdıran program.

C Kodu 

#include <stdio.h>

int main()
{
   int sayi1,sayi2,i;
   unsigned long toplam=0;
   printf("Kucuk ve Buyuk sayiyi giriniz ");
   scanf("%d%d",&sayi1,&sayi2);
   for(i=sayi1;i<=sayi2;++i)
   {
     if(i%2 ==0)
      toplam += i;
   }
   printf("Bu araliktaki cift sayilarin toplami %lu",toplam);
}

Problem Tanımı

[200,300] aralığında rasgele bir sayı üretilsin. Bu sayının kullanıcı tarafından tahmin edilmesi sağlansın.
Kullanıcıya 5 tahmin hakkı sunulsun. 5 tahminde de bulamazsa program sonlandırılsın.
Kullanıcı gizli sayıdan büyük bir değer girerse daha küçük bir sayı girmesi istensin,
Kullanıcı gizli sayıdan küçük bir sayı girerse daha büyük bir sayı girmesi istensin.
Kullanıcı sayıyı tahmin ederse tebrik edip kaçıncı denemede bildiğini ekrana yazdırsın.

C Kodu 

#include <stdio.h>
#include <stdlib.h> //rand , srand

int main()
{
   int gizli_sayi,kullanici_tahmin,tahmin_sayisi=0;
   //her seferde farkli sayi uretmesini sagla
   srand(time(NULL));
   //gizli sayiyi uret [200,300] araliginda
   gizli_sayi = rand()%101 + 200;

   printf("200 ile 300 arasinda sayi girin ");
   do{
       scanf("%d",&kullanici_tahmin);
       tahmin_sayisi++;
       if(kullanici_tahmin>gizli_sayi)
        printf("Daha kucuk sayi girin ");
       else if(kullanici_tahmin<gizli_sayi)
        printf("Daha buyuk sayi girin ");
       else
        printf("Tebrikler %d. denemede bildiniz",tahmin_sayisi);

       if(tahmin_sayisi ==5)
         break;
   }while(kullanici_tahmin!=gizli_sayi);
}

Ekran Çıktısı

 				
200 ile 300 arasinda sayi girin 245
Daha kucuk sayi girin 235
Daha kucuk sayi girin 225
Daha kucuk sayi girin 210
Daha buyuk sayi girin 218
Tebrikler 5. denemede bildiniz

Problem Tanımı

Anti-diyagonalinde + karakteri diğer yerlerde * karakteri bulunan matris yazdırma

C Kodu 

#include <stdio.h>

int main()
{
   int sats,suts,n=7;
   for(sats=1;sats<=n;sats++)
   {
     for(suts=1;suts<=n;suts++)
     {
         if(sats+suts == n+1)
            printf("+");
         else
            printf("*");
     }
     printf("\n");
   }
}

Problem Tanımı

Ana diyagonalinin üst kısmı satır ve sütun numarasının çarpımı,
ana diyagonali ve altı, satır numarasından sütun numarası çıkarılarak oluşturulan matris

C Kodu 

#include <stdio.h>

int main()
{
   int sat_no,sut_no,n=7;
   for(sat_no=1;sat_no<=n;sat_no++)
   {
     for(sut_no=1;sut_no<=n;sut_no++)
     {
         if(sut_no > sat_no)
            printf("%3d",sut_no*sat_no);
         else
            printf("%3d",sat_no-sut_no);
     }
     printf("\n");
   }
}

Problem Tanımı

Yıldızlar kullanarak piramit oluşturma

C Kodu 

#include <stdio.h>
#include <stdlib.h> //rand

int main()
{
   int sat_no,ys,bs,n=7;
   for(sat_no=1;sat_no<=n;sat_no++)
   {
     //bosluk yazdir
     for(bs=1;bs<=n-sat_no;bs++)
     {
       printf(" ");
     }
     //yildiz yazdir
     for(ys=1;ys<=2*sat_no-1;ys++)
     {
       printf("*");
     }
     //ekranda bir alt satira gec
     printf("\n");
   }
}
